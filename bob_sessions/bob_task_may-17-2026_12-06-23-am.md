 57 |

 61 |   generationConfig: {

 66 |   }

 69 | export async function analyzeOverview(repoData: any, files: string[], packageJson: string | null) {
 70 |   const prompt = `
 71 | Analyze this GitHub repository and identify 3-5 logical modules based on the file structure and organization.
 72 | 
 73 | Repository: ${repoData.full_name}
 74 | Description: ${repoData.description || 'No description'}
 75 | Language: ${repoData.language || 'Unknown'}
 76 | Total Files: ${files.length}
 77 | 
 78 | File Structure (first 100 files):
 79 | ${files.slice(0, 100).join('\n')}
 80 | 
 81 | ${packageJson ? `package.json content:\n${packageJson.slice(0, 1000)}` : 'No package.json found'}
 82 | 
 83 | Instructions:
 84 | 1. Identify 3-5 logical modules based on directory structure, file naming, and common patterns
 85 | 2. Each module should represent a cohesive functional area (e.g., Authentication, API, UI Components)
 86 | 3. Assign relevant files to each module (max 15 files per module for display)
 87 | 4. Provide clear, concise descriptions
 88 | 
 89 | Return ONLY valid JSON with this exact structure (no markdown, no explanations, no code blocks):
 90 | {
 91 |   "modules": [
 92 |     {
 93 |       "id": "auth-module",
 94 |       "name": "Authentication Module",
 95 |       "description": "Handles user authentication and authorization",
 96 |       "files": ["src/auth/login.ts", "src/auth/register.ts"],
 97 |       "fileCount": 8
 98 |     }
 99 |   ]
100 | }
101 | `;
102 | 
103 |   try {
104 |     const result = await model.generateContent(prompt);
105 |     const response = result.response.text();
106 |     
107 |     const jsonText = response
108 |       .replace(/```json\n?/g, '')
109 |       .replace(/```\n?/g, '')
110 |       .trim();
111 |     
112 |     const parsed = JSON.parse(jsonText);
113 |     
114 |     if (!parsed.modules || !Array.isArray(parsed.modules)) {
115 |       throw new Error('Invalid response structure: missing modules array');
116 |     }
117 |     
118 |     if (parsed.modules.length === 0) {
119 |       throw new Error('No modules detected in repository');
120 |     }
121 |     
122 |     return parsed;
123 |   } catch (error: any) {
124 |     if (error instanceof SyntaxError) {
125 |       throw new Error('Failed to parse LLM response as JSON. Please try again.');
126 |     }
127 |     throw error;
128 |   }
129 | }

148 | ### Step 2: Update Overview API to Use Real LLM
149 | 
150 | **File**: [`app/api/overview/route.ts`](../app/api/overview/route.ts)
151 | 
152 | **What to do**: Replace mock module generation with real LLM analysis.

154 | **Complete Code**:
155 | 
156 | ```typescript
157 | import { NextResponse } from 'next/server';
158 | import { getRepoOverview } from '@/lib/github';
159 | import { analyzeOverview } from '@/lib/llm';
160 | 
161 | export async function POST(request: Request) {
162 |   try {
163 |     const { owner, repo } = await request.json();
164 |     
165 |     if (!owner || !repo) {
166 |       return NextResponse.json(
167 |         { error: 'Missing owner or repo parameter' },
168 |         { status: 400 }
169 |       );
170 |     }
171 |     
172 |     const { repoData, files, packageJson } = await getRepoOverview(owner, repo);
173 |     
174 |     const analysis = await analyzeOverview(repoData, files, packageJson);
175 |     
176 |     return NextResponse.json({
177 |       repository: {
178 |         owner,
179 |         repo,
180 |         description: repoData.description || 'No description available',
181 |         language: repoData.language,
182 |         stars: repoData.stargazers_count,
183 |         url: repoData.html_url
184 |       },
185 |       modules: analysis.modules,
186 |       totalFiles: files.length,
187 |       hasPackageJson: !!packageJson
188 |     });
189 |   } catch (error: any) {
190 |     console.error('Overview API error:', error);
191 |     
192 |     let errorMessage = 'Failed to analyze repository';
193 |     let statusCode = 500;
194 |     
195 |     if (error.message.includes('not found')) {
196 |       errorMessage = error.message;
197 |       statusCode = 404;
198 |     } else if (error.message.includes('rate limit')) {
199 |       errorMessage = error.message;
200 |       statusCode = 429;
201 |     } else if (error.message.includes('parse')) {
202 |       errorMessage = 'AI analysis failed. Please try again.';
203 |       statusCode = 500;
204 |     }
205 |     
206 |     return NextResponse.json(
207 |       { error: errorMessage },
208 |       { status: statusCode }
209 |     );
210 |   }
211 | }
212 | ```

463 | ## Troubleshooting
464 | 
465 | ### Issue: "Failed to parse LLM response as JSON"
466 | 
467 | **Cause**: Gemini returned markdown-wrapped JSON or invalid JSON
468 | 
469 | **Solution**:
470 | 1. Check server console logs for raw LLM response
471 | 2. Verify prompt is clear about JSON-only output
472 | 3. The code already strips markdown, but if issues persist, add more aggressive cleaning:
473 |    ```typescript
474 |    const jsonText = response
475 |      .replace(/```json\n?/g, '')
476 |      .replace(/```\n?/g, '')
477 |      .replace(/^[^{]*/g, '')  // Remove leading non-JSON
478 |      .replace(/[^}]*$/g, '')  // Remove trailing non-JSON
479 |      .trim();
480 |    ```
481 | 
482 | ### Issue: "No modules detected"
483 | 
484 | **Cause**: LLM returned empty modules array
485 | 
486 | **Solutions**:
487 | 1. Check if repository has sufficient files (needs at least 10-20 files)
488 | 2. Verify file filtering in [`lib/github.ts`](../lib/github.ts) isn't too aggressive
489 | 3. Check if repository is mostly binary files (images, PDFs, etc.)
490 | 
491 | ### Issue: API Timeout
492 | 
493 | **Cause**: Large repository or slow LLM response
494 | 
495 | **Solutions**:
496 | 1. Reduce file list sent to LLM (currently 100 files, can reduce to 50)
497 | 2. Add timeout to LLM call:
498 |    ```typescript
499 |    const timeoutPromise = new Promise((_, reject) => 
500 |      setTimeout(() => reject(new Error('LLM timeout')), 30000)
501 |    );
502 |    const result = await Promise.race([
503 |      model.generateContent(prompt),
504 |      timeoutPromise
505 |    ]);
506 |    ```
507 | 
508 | ### Issue: Rate Limit Errors
509 | 
510 | **Cause**: Too many requests to GitHub API

export async function analyzeOverview(repoData: any, files: string[], packageJson: string | null) {
  const prompt = `
Analyze this GitHub repository and identify 3-5 logical modules based on the file structure and organization.

Repository: ${repoData.full_name}
Description: ${repoData.description || 'No description'}
Language: ${repoData.language || 'Unknown'}
Total Files: ${files.length}

File Structure (first 100 files):
${files.slice(0, 100).join('\n')}

${packageJson ? `package.json content:\n${packageJson.slice(0, 1000)}` : 'No package.json found'}

Instructions:
1. Identify 3-5 logical modules based on directory structure, file naming, and common patterns
2. Each module should represent a cohesive functional area (e.g., Authentication, API, UI Components)
3. Assign relevant files to each module (max 15 files per module for display)
4. Provide clear, concise descriptions

Return ONLY valid JSON with this exact structure (no markdown, no explanations, no code blocks):
{
  "modules": [
    {
      "id": "auth-module",
      "name": "Authentication Module",
      "description": "Handles user authentication and authorization",
      "files": ["src/auth/login.ts", "src/auth/register.ts"],
      "fileCount": 8
    }
  ]
}
`;

  try {
    const result = await model.generateContent(prompt);
    const response = result.response.text();
    
    const jsonText = response
      .replace(/```json\n?/g, '')
      .replace(/```\n?/g, '')
      .trim();
    
    const parsed = JSON.parse(jsonText);
    
    if (!parsed.modules || !Array.isArray(parsed.modules)) {
      throw new Error('Invalid response structure: missing modules array');
    }
    
    if (parsed.modules.length === 0) {
      throw new Error('No modules detected in repository');
    }
    
    return parsed;
  } catch (error: any) {
    if (error instanceof SyntaxError) {
      throw new Error('Failed to parse LLM response as JSON. Please try again.');
    }
    throw error;
  }
}

<content>import { NextResponse } from 'next/server';
import { getRepoOverview } from '@/lib/github';
import { analyzeOverview } from '@/lib/llm';

export async function POST(request: Request) {
  try {
    const { owner, repo } = await request.json();
    
    if (!owner || !repo) {
      return NextResponse.json(
        { error: 'Missing owner or repo parameter' },
        { status: 400 }
      );
    }
    
    const { repoData, files, packageJson } = await getRepoOverview(owner, repo);
    
    const analysis = await analyzeOverview(repoData, files, packageJson);
    
    return NextResponse.json({
      repository: {
        owner,
        repo,
        description: repoData.description || 'No description available',
        language: repoData.language,
        stars: repoData.stargazers_count,
        url: repoData.html_url
      },
      modules: analysis.modules,
      totalFiles: files.length,
      hasPackageJson: !!packageJson
    });
  } catch (error: any) {
    console.error('Overview API error:', error);
    
    let errorMessage = 'Failed to analyze repository';
    let statusCode = 500;
    
    if (error.message.includes('not found')) {
      errorMessage = error.message;
      statusCode = 404;
    } else if (error.message.includes('rate limit')) {
      errorMessage = error.message;
      statusCode = 429;
    } else if (error.message.includes('parse')) {
      errorMessage = 'AI analysis failed. Please try again.';
      statusCode = 500;
    }
    
    return NextResponse.json(
      { error: errorMessage },
      { status: statusCode }
    );
  }
}
